**Extend `frontend/src/hooks/useFolderTree.ts`** with mutation methods:

Add these functions (all use session.access_token for auth, then call refreshTree + reload contents):

```typescript
const createFolder = useCallback(async (name: string, parentId: string | null) => {
  await fetch('/api/folders', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json', Authorization: `Bearer ${session.access_token}` },
    body: JSON.stringify({ name, parent_id: parentId }),
  })
  await refreshTree()
}, [session, refreshTree])

const renameFolder = useCallback(async (folderId: string, name: string) => {
  await fetch(`/api/folders/${folderId}`, {
    method: 'PATCH',
    headers: { 'Content-Type': 'application/json', Authorization: `Bearer ${session.access_token}` },
    body: JSON.stringify({ name }),
  })
  await refreshTree()
  if (selectedFolderId) await loadContents(selectedFolderId)
}, [session, refreshTree, selectedFolderId, loadContents])

const deleteFolder = useCallback(async (folderId: string) => {
  await fetch(`/api/folders/${folderId}`, {
    method: 'DELETE',
    headers: { Authorization: `Bearer ${session.access_token}` },
  })
  if (selectedFolderId === folderId) selectFolder(null)
  await refreshTree()
}, [session, refreshTree, selectedFolderId, selectFolder])

const moveFolder = useCallback(async (folderId: string, newParentId: string | null) => {
  await fetch(`/api/folders/${folderId}/move`, {
    method: 'PATCH',
    headers: { 'Content-Type': 'application/json', Authorization: `Bearer ${session.access_token}` },
    body: JSON.stringify({ new_parent_id: newParentId }),
  })
  await refreshTree()
  if (selectedFolderId) await loadContents(selectedFolderId)
}, [session, refreshTree, selectedFolderId, loadContents])

const moveDocument = useCallback(async (docId: string, folderId: string | null) => {
  await fetch(`/api/documents/${docId}/move`, {
    method: 'PATCH',
    headers: { 'Content-Type': 'application/json', Authorization: `Bearer ${session.access_token}` },
    body: JSON.stringify({ folder_id: folderId }),
  })
  if (selectedFolderId) await loadContents(selectedFolderId)
}, [session, selectedFolderId, loadContents])

const renameDocument = useCallback(async (docId: string, filename: string) => {
  await fetch(`/api/documents/${docId}`, {
    method: 'PATCH',
    headers: { 'Content-Type': 'application/json', Authorization: `Bearer ${session.access_token}` },
    body: JSON.stringify({ filename }),
  })
  if (selectedFolderId) await loadContents(selectedFolderId)
}, [session, selectedFolderId, loadContents])
```
